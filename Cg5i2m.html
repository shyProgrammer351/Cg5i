프로그램 코드

	
	#include <stdio.h>
	
	typedef struct { 	
		char name[16];							
		int korean;
		int english;
		int mathematics;
		double average;
	} Student;  
	
1	int main() {															
2		Student student = { "", 0, 0, 0, 0.0 };					
	
		printf( "학생 이름을 입력하세요: " );
		scanf( "%s", student.name );							
		printf( "0점 ~ 100점 사이의 국어 점수를 입력하세요: " );
		scanf( "%d", &student.korean );							
		printf( "0점 ~ 100점 사이의 영어 점수를 입력하세요: " );
		scanf( "%d", &student.english );							
		printf( "0점 ~ 100점 사이의 수학 점수를 입력하세요: " );
		scanf( "%d", &student.mathematics );						
3		student.average = (double)( student.korean + student.english + student.mathematics ) / 3; 
	
4		printf( "%s 학생은 국어 %d점, 영어 %d점, 수학 %d점, 평균 %.2f점입니다", 
			student.name, student.korean, student.english, student.mathematics, student.average );	
		
5		return 0;
	}
