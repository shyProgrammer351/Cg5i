알고리즘

프로그램 시작 
   우리반 학생수 및 점수를 입력 
      학생수만큼 점수 저장 공간을 확보 
   평균 점수 계산 및 출력 
   점수 저장 공간을 반납 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter15/ClassAverage.c

	#include <stdio.h>

	#include <stdlib.h>

		 

	// 프로그램 시작 

1	int main() {

		// 우리반 학생수 및 점수를 입력 

		printf( "우리반 학생수를 입력하세요: " );

		int count = 0;

2		scanf( "%d", &count );

	 

		// 학생수만큼 점수 저장 공간을 확보 

3		int* scores =

4		              (int*)malloc( sizeof(int) * count );

5		for( int index = 0; index < count; index++ ) {

			printf( "점수를 입력하세요: " );

6			scanf( "%d", &scores[index] );

		}

		 

		// 평균 점수 계산 및 출력  

		int sum = 0;

		for( int index = 0; index < count; index++ ) {

			sum += scores[index];

		}

		 

7		printf( "우리반 학생들의 평균 점수는 %.2f입니다", (double)sum / count );

		 

		// 점수 저장 공간을 반납 

8		free( scores );

		 

		// 프로그램 종료 

9		return 0;

	}
