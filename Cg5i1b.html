문제

캐릭터가 한 칸 이동해서 방패([), 칼(/), 창(->), 출구(문) 중 하나를 선택하는 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요 
      / 
  [   옷   문
      -> 
WASD와 [Enter]를 입력하세요: a
      / 
 옷]       문
      ->


      / 
  [   옷   문
      -> 
WASD와 [Enter]를 입력하세요: W
      옷/ 
  [   옷   문
      ->


      / 
  [   옷   문
      -> 
WASD와 [Enter]를 입력하세요: s
      / 
  [        문
      옷->

코드 빈칸 채우기

# <> 					 
 
# up 'w' 
# UP 'W' 
# down 's' 
# DOWN 'S' 
# left 'a' 
# LEFT 'A' 
# right 'd' 
# RIGHT 'D' 
 
 main() { 
	char direction = UP; 
 
	printf("      / \n"); 
	printf("  [   옷   문\n"); 
	printf("      -> \n\n"); 
	 
	printf("방향키(WASD)와 [Enter]를 입력하세요: " ); 
	scanf("%c",&direction); 
 
	printf(" [2;6f  "); 
	( direction ) { 
		 UP:  up:  
			printf(" [1;6f 옷/ "); 
			; 
		 LEFT:  left:  
			printf(" [2;1f 옷] "); 
			; 
		 DOWN:  down:  
			printf(" [3;6f 옷-> "); 
			; 
		 RIGHT:  right:  
			printf(" [1;1f              \n"); 
			printf("   Success!!   \n"); 
			printf("              "); 
	} 
 
	return 0; 
}
