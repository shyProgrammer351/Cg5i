문제

캐릭터가 한 칸 이동해서 방패([), 칼(/), 창(->), 출구(문) 중 하나를 선택하는 문제입니다 실행순서를 클릭하세요 
      / 
  [   옷   문
      -> 
WASD와 [Enter]를 입력하세요: a
      / 
 옷]       문
      ->

프로그램 코드

	#include <stdio.h>


	#define up 'w'

	#define UP 'W'

	#define down 's'

	#define DOWN 'S'

	#define left 'a'

	#define LEFT 'A'

	#define right 'd'

	#define RIGHT 'D'


1	int main() {

2		char direction = UP;


3		printf("      / \n");

4		printf("  [   옷   문\n");

5		printf("      -> \n\n");

		

6		printf("방향키(WASD)와 [Enter]를 입력하세요: " );

7		scanf("%c",&direction);


8		printf(" \033[2;6f  ");

9		switch( direction ) {

10			case UP: case up:

11				printf(" \033[1;6f 옷/ ");

12				break;

13			case LEFT: case left:

14				printf(" \033[2;1f 옷] ");

15				break;

16			case DOWN: case down:

17				printf(" \033[3;6f 옷-> ");

18				break;

19			case RIGHT: case right:

20				printf(" \033[1;1f              \n");

21				printf("   Success!!   \n");

22				printf("              ");

		}


23		return 0;

	}
