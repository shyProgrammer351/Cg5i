 char map[100][100]; 
 
int readMap( char *filename ) { 
	FILE *inFile = NULL; 
 
	 ( ( inFile = fopen( filename, "rt" ) ) == NULL ) { 
		printf( "[오류] 파일을 열 수 없습니다!" ); 
		return -1; 
	} 
 
	( int row=0; fgets( map[row], 10, inFile ) != NULL; row++ ) { 
		 ( map[row][0] == '\n' ) { 
			row--; 
		} 
	} 
 
 
	fclose( inFile ); 
	return 0; 
} 
 
void writeMap() { 
	printf("\033[1;1f                                                                                       \033[1;1f"); 
	( int row = 0; map[row][0] != '\0'; row++ ) { 
		( int column = 0; map[row][column] != '\0'; column++ ) { 
			( map[row][column] == 'W' ) 
				printf("\033[44m  \033[0m"); 
			( map[row][column] == ' ' ) 
				printf("  "); 
			( map[row][column] == 'E' ) 
				printf("\033[34m문\033[0m"); 
			( map[row][column] == 'A' ) 
				printf("\033[31m옷\033[0m"); 
		} 
		printf("\n"); 
	} 
}

void move( char direction, int *row, int *column ) { 
	 ( ( map[ *row-1 ][ *column ] != 'W' ) && ( ( direction == 'w' ) || ( direction == 'W' ) ) )  
		*row = *row-1; 
	 ( ( map[ *row ][ *column-1 ] != 'W' ) && ( ( direction == 'a' ) || ( direction == 'A' ) ) )  
		*column = *column-1; 
	 ( ( map[ *row+1 ][ *column ] != 'W' ) && ( ( direction == 's' ) || ( direction == 'S' ) ) )  
		*row = *row+1; 
	 ( ( map[ *row ][ *column+1 ] != 'W' ) && ( ( direction == 'd' ) || ( direction == 'D' ) ) )  
		*column = *column+1; 
}
