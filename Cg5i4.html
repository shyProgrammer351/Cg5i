알고리즘

프로그램시작 
   3개 컵 중에서 하나에 구슬 숨기기 
   사용자에게 구슬을 숨긴 컵 번호 입력 받기 
   구슬 찾기 결과 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter15/ThreeCupGame.c

	#include <stdio.h>

	#include <stdlib.h>

	#include <time.h>

	 

	#define DEBUG

	 

	// 프로그램시작  

1	int main() {

		int ball=0, cup=0;

		 

		// 3개 컵 중에서 하나에 구슬 숨기기 

		srand( time( NULL ) );

2		ball = rand() % 3 + 1;

		// 디버깅 모드에서는 구슬 위치를 공개

3		#ifdef DEBUG

4		printf("\033[8;1f [DEBUG] 구슬 위치 %d \033[1;1f", ball);

5		#endif

		 

		printf("  ___    ___    ___  \n");

		printf(" |   |  |   |  |   | \n");

		printf(" | 1 |  | 2 |  | 3 | \n\n\n\n");

	 

		// 사용자에게 구슬을 숨긴 컵 번호 입력 받기  

		printf(" 1, 2, 3 중에서 구슬[O]을 숨긴 컵을 찾으세요 : ");

6		scanf( "%d", &cup );

			 

		// 구슬 찾기 결과 출력 

		printf(" \033[4;%dfO\n \033[5;%df", ball*7-3, cup*7-5 );

7		( cup == ball ) ? printf("찾았다!\n") : printf("놓쳤다!\n");

	 

		// 프로그램 종료 

8		return 0;

	}
